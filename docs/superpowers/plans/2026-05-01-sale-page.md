```js
async function handleSaleRotate(request, env) {
  if (!isAdmin(request, env)) return jsonRes({ error: 'Unauthorized' }, 401, request);

  const now = new Date().toISOString();
  const nextRotation = new Date(Date.now() + 7 * 24 * 60 * 60 * 1000).toISOString();

  // Clear current sale
  await env.DB.prepare('UPDATE photos SET on_sale=0, sale_started_at=NULL WHERE on_sale=1').run();

  // Pick 50 random published photos
  const { results } = await env.DB.prepare(
    'SELECT id FROM photos WHERE published=1 ORDER BY RANDOM() LIMIT 50'
  ).all();

  if (results.length === 0) return jsonRes({ error: 'No published photos found' }, 400, request);

  // Set them on sale
  for (const photo of results) {
    await env.DB.prepare(
      'UPDATE photos SET on_sale=1, sale_started_at=? WHERE id=?'
    ).bind(now, photo.id).run();
  }

  return jsonRes({ ok: true, rotated: results.length, next_rotation: nextRotation }, 200, request);
}
```

```html
const SITE_URL = 'https://amitphotos.com';
const SALE_DISCOUNT = 0.20;
const BASE_PRICES = { small: 19, medium: 59, large: 129 };

async function loadPrices() {
  try {
    const d = await fetch('/api/admin/prices').then(r => r.ok ? r.json() : null);

function salePrice(size) {
  return Math.round(BASE_PRICES[size] * (1 - SALE_DISCOUNT));

function startCountdown(saleStartedAt) {
  const saleEnd = new Date(saleStartedAt);
  saleEnd.setDate(saleEnd.getDate() + 7);

  function tick() {
    const diff = saleEnd - Date.now();
    if (diff <= 0) { document.getElementById('countdown').textContent = 'המבצע הסתיים'; return; }
    const days  = Math.floor(diff / 86400000);
    const hours = Math.floor((diff % 86400000) / 3600000);
    const mins  = Math.floor((diff % 3600000)  / 60000);
    document.getElementById('cd-days').textContent  = String(days).padStart(2,'0');
    document.getElementById('cd-hours').textContent = String(hours).padStart(2,'0');
    document.getElementById('cd-mins').textContent  = String(mins).padStart(2,'0');
  }
  tick();
  setInterval(tick, 30000);
}
```
